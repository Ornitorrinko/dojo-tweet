const get = async (hashtag) => {
  const params = {
    q: `#${hashtag}`,
    count: 10,
    resultType: 'recent',
    include_entities: false
  };
 
  return client.get('search/tweets.json', params)
    .then((results) => {
      return results.statuses.map(item => item.text)
    })
}
